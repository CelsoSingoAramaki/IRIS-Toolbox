1 function varargout = prepareSteady(this, displayMode, varargin)
  2 % prepareSteady  Prepare steady-state solver.
  3 %
  4 % Backend IRIS function.
  5 % No help provided.
  6 
  7 % -IRIS Macroeconomic Modeling Toolbox.
  8 % -Copyright (c) 2007-2017 IRIS Solutions Team.
  9 
 10 % Run user-supplied steady-state solver:
 11 % sstate = @func
 12 if length(varargin)==1 && isa(varargin{1}, 'function_handle')
 13     varargout{1} = varargin{1};
 14     return
 15 end
 16 
 17 % Run user-supplied steady-state solver with extra arguments:
 18 % sstate = { @func, arg2, arg3,...}
 19 if length(varargin)==1 && iscell(varargin{1}) ...
 20         && ~isempty(varargin{1}) ...
 21         && isa(varargin{1}{1}, 'function_handle')
 22     varargout{1} = varargin{1};
 23     return
 24 end
 25 
 26 % Do not run steady-state solver:
 27 % sstate = false
 28 if length(varargin)==1 && isequal(varargin{1}, false)
 29     varargout{1} = false;
 30     return
 31 end
 32 
 33 % Do run steady-state solve with default options:
 34 % sstate = true
 35 if length(varargin)==1 && isequal(varargin{1}, true)
 36     varargin(1) = [ ];
 37 end
 38 
 39 %--------------------------------------------------------------------------
 40 
 41 
 42 if this.IsLinear
 43     % Linear steady state solver
 44     %----------------------------
 45     opt = passvalopt('model.SteadyLinear', varargin{:});
 46     varargout{1} = opt;
 47 else
 48     % Nonlinear steady state solver
 49     %-------------------------------
 50     % Capture obsolete syntax with solver options directly passed among other
 51     % sstate options and not as suboptions through Solver=; these are only used
 52     % if Solver= is a char.
 53     [opt, obsoleteSolverOpt] = passvalopt('model.SteadyNonlinear', varargin{:});
 54     if isequal(opt.Growth, @auto)
 55         opt.Growth = this.IsGrowth;
 56     end
 57     [opt.Solver, opt.PrepareGradient] = ...
 58         solver.Options.processOptions(opt.Solver, opt.PrepareGradient, displayMode, obsoleteSolverOpt);
 59     blz = createBlocks(this, opt);
 60     blz = processFixOpt(this, blz, opt);
 61     blz = prepareBounds(this, blz, opt);
 62     blz.NanInit = opt.NanInit;
 63     blz.Reuse = opt.Reuse;
 64     blz.Warning = opt.Warning;
 65     varargout{1} = blz;
 66 end
 67 
 68 end

 73 function blz = processFixOpt(this, blz, opt)
 74 % Process the fix, fixallbut, fixlevel, fixlevelallbut, fixgrowth,
 75 % and fixgrowthallbut options. All the user-supply information is
 76 % combined into fixlevel and fixgrowth.
 77 TYPE = @int8;
 78 PTR = @int16;
 79 
 80 nQty = length(this.Quantity.Name);
 81 ixy = this.Quantity.Type==TYPE(1);
 82 ixx = this.Quantity.Type==TYPE(2);
 83 ixp = this.Quantity.Type==TYPE(4);
 84 ixCanBeFixed =  ixy | ixx | ixp;
 85 list = {'fix', 'fixlevel', 'fixgrowth'};
 86 for i = 1 : length(list)
 87     fix = list{i};
 88     fixAllBut = [fix, 'allbut'];
 89 
 90     % Convert charlist to cellstr.
 91     if ischar(opt.(fix)) && ~isempty(opt.(fix))
 92         opt.(fix) = regexp(opt.(fix), '\w+', 'match');
 93     end
 94     if ischar(opt.(fixAllBut)) && ~isempty(opt.(fixAllBut))
 95         opt.(fixAllBut) = ...
 96             regexp(opt.(fixAllBut), '\w+', 'match');
 97     end
 98 
 99     % Convert fixAllBut to fix.
100     if ~isempty(opt.(fixAllBut))
101         lsNameCanBeFixed = this.Quantity.Name(ixCanBeFixed);
102         opt.(fix) = setdiff(lsNameCanBeFixed, opt.(fixAllBut));
103     end
104 
105     if ~isempty(opt.(fix))
106         ell = lookup(this.Quantity, opt.(fix), ...
107             TYPE(1), ...
108             TYPE(2), ...
109             TYPE(4));
110         posFix  = ell.PosName;
111         ixValid = ~isnan(posFix);
112         if any(~ixValid)
113             throw( exception.Base('Steady:CANNOT_FIX', 'error'), opt.(fix){~ixValid} );
114         end
115         opt.(fix) = posFix;
116     else
117         opt.(fix) = [ ];
118     end
119 end
120 
121 fixL = false(1, nQty);
122 fixL(opt.fix) = true;
123 fixL(opt.fixlevel) = true;
124 fixG = false(1, nQty);
125 % Fix growth of endogenized parameters to zero.
126 fixG(ixp) = true;
127 if opt.Growth
128     fixG(opt.fix) = true;
129     fixG(opt.fixgrowth) = true;
130 else
131     fixG(:) = true;
132 end
133 % Fix optimal policy multipliers. The level and growth of
134 % multipliers will be set to zero in the main loop.
135 if opt.zeromultipliers
136     fixL = fixL | this.Quantity.IxLagrange;
137     fixG = fixG | this.Quantity.IxLagrange;
138 end
139 
140 blz.PosFix = struct( );
141 blz.PosFix.Level = PTR( find(fixL) ); %#ok<FNDSB>
142 blz.PosFix.Growth = PTR( find(fixG) ); %#ok<FNDSB>
143 
144 % Remove quantities fixed by user and LHS quantities from dynamic links.
145 temp = getActiveLhsPtr(this.Link);
146 posExclude = struct( );
147 posExclude.Level = [blz.PosFix.Level, temp];
148 posExclude.Growth = [blz.PosFix.Growth, temp];
149 exclude(blz, posExclude);
150 end

155 function blz = createBlocks(this, opt)
156 TYPE = @int8;
157 
158 nQty = length(this.Quantity.Name);
159 ixe = this.Quantity.Type==TYPE(31) | this.Quantity.Type==TYPE(32);
160 ixp = this.Quantity.Type==TYPE(4);
161 
162 % Run solver.blazer.Blazer on steady equations.
163 blz = prepareBlazer(this, 'steady', opt);
164 run(blz);
165 
166 % Prepare solver.block.Blocks for evaluation.
167 prepareBlocks(blz, opt);
168 
169 if blz.IsSingular
170     throw( ...
171         exception.Base('Steady:STRUCTURAL_SINGULARITY', 'warning')...
172         );
173 end
174 
175 % Index of level variables that will be always set to zero.
176 ixZero = struct( );
177 ixZero.Level = false(1, nQty);
178 ixZero.Level(ixe) = true;
179 if opt.Growth
180     ixZero.Growth = false(1, nQty);
181     ixZero.Growth(ixe | ixp) = true;
182 else
183     ixZero.Growth = true(1, nQty);
184 end
185 if opt.zeromultipliers
186     ixZero.Level = ixZero.Level | this.Quantity.IxLagrange;
187     ixZero.Growth = ixZero.Growth | this.Quantity.IxLagrange;
188 end
189 blz.IxZero = ixZero;
190 end

195 function blz = prepareBounds(this, blz, opt)
196 nQty = length(this.Quantity.Name);
197 nBlk = length(blz.Block);
198 ixValidLevel = true(1, nQty);
199 ixValidGrowth = true(1, nQty);
200 for iBlk = 1 : nBlk
201     blk = blz.Block{iBlk};
202     if blk.Type~=solver.block.Type.SOLVE
203         continue
204     end
205 
206     % Level bounds.
207     lsLevel = this.Quantity.Name(blk.PosQty.Level);
208     [lbl, ubl, ixValidLevel] = ...
209         boundsHere(lsLevel, blk.PosQty.Level, opt.levelbounds, ixValidLevel);
210 
211     % Growth bounds.
212     lsGrowth = this.Quantity.Name(blk.PosQty.Growth);
213     [lbg, ubg, ixValidGrowth] = ...
214         boundsHere(lsGrowth, blk.PosQty.Growth, opt.growthbounds, ixValidGrowth);
215 
216     % Combine level and growth bounds
217     blk.Lower = [lbl, lbg];
218     blk.Upper = [ubl, ubg];
219 
220     if isa(blk.Solver, 'optim.options.SolverOptions')
221         % Make sure @lsqnonlin is used when there are some lower/upper bounds.
222         isBnd = any(~isinf(blk.Lower)) || any(~isinf(blk.Upper));
223         if isBnd
224             if ~strcmp(blk.Solver.SolverName, 'lsqnonlin')
225                 blk.Solver = blk.Solver('lsqnonlin', blk.Solver);
226             end
227         end
228     end
229 
230     blz.Block{iBlk} = blk;
231 end
232 
233 if any(~ixValidLevel)
234     throw( exception.Base('Steady:WRONG_SIGN_LEVEL_BOUNDS', 'error'), ...
235         this.Quantity.Name{~ixValidLevel} );
236 end
237 if any(~ixValidGrowth)
238     throw( exception.Base('Steady:WRONG_SIGN_GROWTH_BOUNDS', 'error'), ...
239         this.Quantity.Name{~ixValidGrowth} );
240 end
241 
242 return

247     function [vecLb, vecUb, ixValid] = boundsHere(list, nameId, bnd, ixValid)
248         nList = length(list);
249         vecLb = -inf(1, nList);
250         vecUb = inf(1, nList);
251         for jj = 1 : nList
252             name = list{jj};
253             %{
254             isLogPlus = opt.IxLogPlus( nameId(jj) );
255             isLogMinus = opt.IxLogMinus( nameId(jj) );
256             %}
257             isLog = blz.IxLog( nameId(jj) );
258             %{
259             try
260                 lb = bnd.(name)(1);
261             catch
262                 if isLogPlus
263                     lb = 0;
264                 else
265                     lb = -Inf;
266                 end
267             end
268             try
269                 ub = bnd.(name)(2);
270             catch
271                 if isLogMinus
272                     ub = 0;
273                 else
274                     ub = Inf;
275                 end
276             end
277             %}
278             try
279                 lb = bnd.(name)(1);
280             catch
281                 if isLog
282                     lb = 0;
283                 else
284                     lb = -Inf;
285                 end
286             end
287             try
288                 ub = bnd.(name)(2);
289             catch
290                 ub = Inf;
291             end
292             %{
293             if isLogPlus
294                 lb = log(lb);
295                 ub = log(ub);
296             elseif isLogMinus
297                 % Swap lower and upper bounds for log-minus variables.
298                 ub = log(-lb);
299                 lb = log(-ub);
300             end
301             %}
302             if isLog
303                 lb = log(lb);
304                 ub = log(ub);
305             end
306             if imag(lb)==0 && imag(ub)==0
307                 % Assign only if both lower and upper bound is ok.
308                 vecLb(jj) = lb;
309                 vecUb(jj) = ub;
310             else
311                 % Report problem for this variables.
312                 ixValid( nameId(jj) ) = false;
313             end
314         end
315     end
316 end
