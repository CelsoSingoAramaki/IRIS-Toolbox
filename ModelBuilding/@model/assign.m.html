1 function [this, lsAssigned] = assign(this, varargin)
  2 % assign  Assign parameters, steady states, std deviations or cross-correlations
  3 %
  4 %
  5 % __Syntax__
  6 %
  7 %     [M, Assigned] = assign(M, P)
  8 %     [M, Assigned] = assign(M, N)
  9 %     [M, Assigned] = assign(M, Name, Value, Name, Value, ...)
 10 %     [M, Assigned] = assign(M, List, Values)
 11 %
 12 %
 13 % __Syntax for Fast Assign__
 14 %
 15 %     % Initialise
 16 %     assign(M, List);
 17 %
 18 %     % Fast assign
 19 %     M = assign(M, Values);
 20 %     ...
 21 %     M = assign(M, Values);
 22 %     ...
 23 %
 24 %
 25 % __Input arguments__
 26 %
 27 % * `M` [ model ] - Model object.
 28 %
 29 % * `P` [ struct ] - Database whose fields refer to parameter
 30 % names, variable names, std deviations, or cross-correlations.
 31 %
 32 % * `N` [ model ] - Another model object from which all parameteres
 33 % (including std erros and cross-correlation coefficients), and
 34 % steady-states values will be assigned that match the name and type in
 35 % `M`.
 36 %
 37 % * `Name` [ char ] - A parameter name, variable name, std
 38 % deviation, cross-correlation, or a regular expression that will be
 39 % matched against model names.
 40 %
 41 % * `Value` [ numeric ] - A value (or a vector of values in case of
 42 % multiple parameterisations) that will be assigned.
 43 %
 44 % * `List` [ cellstr ] - A list of parameter names, variable names, std
 45 % deviations, or cross-correlations.
 46 %
 47 % * `Values` [ numeric ] - A vector of values.
 48 %
 49 %
 50 % __Output arguments__
 51 %
 52 % * `M` [ model ] - Model object with newly assigned parameters and/or
 53 % steady states.
 54 %
 55 % * `Assigned` [ cellstr | `Inf` ] - List of actually assigned parameter
 56 % names, variables names (steady states), std deviations, and
 57 % cross-correlations; `Inf` indicates that all values has been assigned
 58 % from another model object.
 59 %
 60 %
 61 % __Description__
 62 %
 63 % Calls with `Name`-`Value` or `List`-`Value` pairs throw an error if some
 64 % names in the list are not valid names in the model object. Calls with a
 65 % database, `P`, or another model object, `N`, do not perform this check.
 66 %
 67 %
 68 % __Example__
 69 %
 70 
 71 % -IRIS Macroeconomic Modeling Toolbox.
 72 % -Copyright (c) 2007-2017 IRIS Solutions Team.
 73 
 74 persistent POS_QTY IX_QTY_RHS POS_STDCORR IX_STDCORR_RHS LS_ASSIGNED;
 75 TYPE = @int8;
 76 
 77 %--------------------------------------------------------------------------
 78 
 79 ixe = this.Quantity.Type==TYPE(31) | this.Quantity.Type==TYPE(32);
 80 
 81 lsAssigned = cell(1, 0);
 82 if isempty(varargin)
 83     return
 84 end
 85 
 86 flags = struct('Level', false, 'Growth', false, 'ExcludeNaNs', false);
 87 readFlags( );
 88 
 89 % Number of input arguments with the growth label removed.
 90 n = length(varargin);
 91 ne = sum(ixe);
 92 nStdCorr = ne + ne*(ne-1)/2;
 93 nAlt = length(this.Variant);
 94 nQty = length(this.Quantity);
 95 lsAllName = cell(1, 0);
 96 lsInvalidLen = cell(1, 0);
 97 lsInvalidImag = cell(1, 0);
 98 lsInvalidLhsName = cell(1, 0);
 99 
100 % `Assign` and `stdcorr` are logical indices of values that have been
101 % assigned.
102 ixQty = false(1, nQty);
103 ixStdCorr = false(1, nStdCorr);
104 
105 if isempty(varargin)
106     % Do nothing.
107 
108 elseif isa(varargin{1}, 'model')
109     % Assign from another model object:
110     % m = assign(m, n);
111     % m = assign(m, n, list);
112     assignFromModelObj( );
113 
114 elseif n==1 && isnumeric(varargin{1})
115     % Quick assignment after iniatialization.
116     % m = assign(m,array).
117     if isempty(POS_QTY) && isempty(POS_STDCORR)
118         utils.error('modelobj:assign', ...
119             ['Initialize assign( ) before using the function ', ...
120             'with a single numeric input.']);
121     end
122     ixQty(POS_QTY) = true;
123     ixStdCorr(POS_STDCORR) = true;
124 
125     this.Variant = model.Variant.assignQuantity( ...
126         this.Variant, POS_QTY, ':', varargin{1}(IX_QTY_RHS) ...
127         );
128 
129     this.Variant = model.Variant.assignStdCorr( ...
130         this.Variant, POS_STDCORR, ':', varargin{1}(IX_STDCORR_RHS), ...
131         this.Quantity.IxStdCorrAllowed ...
132         );
133 
134     lsAssigned = LS_ASSIGNED;
135 
136     % Keep persistent variables and return immediately.
137     return
138 
139 elseif n<=2 && iscellstr(varargin{1})
140     % assign(m, cellstr) initializes quick-assign function.
141     % m = assign(m, cellstr, array);
142     lsAllName = varargin{1}(:).';
143     varargin(1) = [ ];
144     nName = length(lsAllName);
145     ell = lookup(this.Quantity, lsAllName);
146     posQty = ell.PosName;
147     posStdCorr = ell.PosStdCorr;
148     ixQtyRhs = ~isnan(posQty);
149     posQty = posQty(ixQtyRhs);
150     ixStdCorrRhs = ~isnan(posStdCorr);
151     posStdCorr = posStdCorr(ixStdCorrRhs);
152     lsInvalidLhsName = lsAllName(~ixQtyRhs & ~ixStdCorrRhs);
153 
154     if isempty(varargin)
155         % Initialize quick-assign access and return.
156         POS_QTY = posQty;
157         POS_STDCORR = posStdCorr;
158         IX_QTY_RHS = ixQtyRhs;
159         IX_STDCORR_RHS = ixStdCorrRhs;
160         listAssigned( );
161         LS_ASSIGNED = lsAssigned;
162 
163         % Keep persistent variables and return immediately.
164         return
165     elseif isnumeric(varargin{1})
166         value = varargin{1};
167         assignListAndValue( );
168         chkLhsNames( );
169     else
170         utils.error('modelobj:assign', '#Invalid_assign:model');
171     end
172 
173 elseif n<=2 && isstruct(varargin{1})
174     % Assign from a database:
175     % m = assign(m, struct);
176     % m = assign(m, struct, clone);
177     d = varargin{1};
178     varargin(1) = [ ];
179     lsAllName = fieldnames(d);
180     allValue = struct2cell(d);
181     if ~isempty(varargin) && ~isempty(varargin{1})
182         cloneTemplate = varargin{1};
183         lsAllName = parser.Preparser.cloneAllNames(lsAllName, cloneTemplate);
184     end
185     lsAllName = lsAllName(:).';
186     allValue = allValue(:).';
187     nName = length(lsAllName);
188     lsInvalidLen = cell(1, 0);
189     lsInvalidImag = cell(1, 0);
190     ell = lookup(this.Quantity, lsAllName);
191     posQty = ell.PosName;
192     posStdCorr = ell.PosStdCorr;
193     for i = 1 : nName
194         name = lsAllName{i};
195         value = allValue{i};
196         % Do not check if names are valid LHS names in assignments from databases.
197         if isnan(posQty(i)) && isnan(posStdCorr(i))
198             continue
199         end
200         assignNameAndValue(name, posQty(i), posStdCorr(i), value);
201     end
202     chkValid( );
203 
204 elseif all(cellfun(@(x) ischar(x) || isa(x, 'rexp'), varargin(1:2:end)))
205     % m = assign(m, name, value, name, value, ...)
206     % name is char or char list or rexp or double dot but not cellstr.
207     lsAllName = varargin(1:2:end);
208     nName = length(lsAllName);
209     % Remove equal signs from assign(m, 'alpha=', 1).
210     for i = 1 : nName
211         lsAllName{i} = strrep(lsAllName{i},' ','');
212         if ~isempty(lsAllName{i}) && lsAllName{i}(end)=='='
213             lsAllName{i}(end) = '';
214         end
215     end
216     allValue = varargin(2:2:end);
217     lsInvalidLhsName = cell(1, 0);
218     lsInvalidLen = cell(1, 0);
219     lsInvalidImag = cell(1, 0);
220     for i = 1 : nName
221         name = lsAllName{i};
222         value = allValue{i};
223         if isa(name, 'rexp') || isempty(regexp(name, '\W', 'once'))
224             % Plain name.
225             assignNameAndValue(name, [ ], [ ], allValue{i});
226         else
227             % List of names:
228             % A1, A2, B
229             % or double-dot list 
230             % A1,..,A2
231             list = name;
232             if ~isempty(strfind(list, ',..,'))
233                 list = parse(parser.doubledot.Keyword.COMMA, list);
234             end
235             list = regexp(list, '\w+', 'match');
236             for j = 1 : length(list)
237                 ddName = list{j};
238                 if size(value, 2)==1
239                     ddValue = value;
240                 else
241                     ddValue = value(:, j, :);
242                 end
243                 assignNameAndValue(ddName, [ ], [ ], ddValue);
244             end
245         end
246     end
247     chkValid( );
248     chkLhsNames( );
249 
250 else
251     % Throw an invalid assignment error.
252     throw( exception.Base('General:INVALID_ASSIGNMENT', 'error'), class(this) ); %#ok<GTARG>
253 end
254 
255 % Reset persistent variables in each non-quick-assign calls.
256 POS_QTY = [ ];
257 IX_QTY_RHS = [ ];
258 POS_STDCORR = [ ];
259 IX_STDCORR_RHS = [ ];
260 LS_ASSIGNED = { };
261 
262 % Steady states cannot be changed from 0+0i.
263 ixZeroShock = true(1, ne);
264 for iAlt = 1 : nAlt
265     ixZeroShock = ixZeroShock & this.Variant{iAlt}.Quantity(ixe)==0;
266 end
267 if any(~ixZeroShock)
268     temp = true(1, nQty);
269     temp(ixe) = ixZeroShock;
270     throw( ...
271         exception.Base('Model:CANNOT_CHANGE_STEADY_SHOCKS', 'error'), ...
272         this.Quantity.Name{~temp} ...
273         );
274 end
275 
276 if nargout<2
277     return
278 end
279 
280 % Put together list of parameters, steady states, std deviations, and
281 % correlations that have been assigned.
282 listAssigned( );
283 
284 return

289     function readFlags( )
290         while ~isempty(varargin) && ischar(varargin{1}) ...
291                 && strncmp(varargin{1}, '-', 1)
292             switch lower(strtrim(varargin{1}))
293                 case '-level'
294                     flags.Level = true;
295                 case '-growth'
296                     flags.Growth = true;
297                 case '-excludenans'
298                     flags.ExcludeNaNs = true;
299                 otherwise
300                     throw( exception.Base('Model:INVALID_ASSIGN_FLAG', 'error'), ...
301                         varargin{1} );
302             end
303             varargin(1) = [ ];
304         end
305     end

310     function assignFromModelObj( )
311         rhs = varargin{1};
312         lsToAssign = @all;
313         if n>1
314             lsToAssign = varargin{2};
315             if ischar(lsToAssign)
316                 lsToAssign = regexp(lsToAssign, '\w+', 'match');
317             end
318         end
319         nAltRhs = length(rhs.Variant);
320         if nAltRhs~=1 && nAltRhs~=nAlt
321             utils.error('modelobj:assign', ...
322                 ['Cannot assign from object ', ...
323                 'with different number of parameterisations.']);
324         end
325         nQty = length(this.Quantity);
326         ixMatchingType = true(1, nQty);
327         for ii = 1 : nQty
328             name = this.Quantity.Name{ii};
329             if ~isequal(lsToAssign, @all) && ~any(strcmpi(name, lsToAssign))
330                 continue
331             end
332             ixRhs = strcmp(name, rhs.Quantity.Name);
333             if ~any(ixRhs)
334                 continue
335             end
336             if rhs.Quantity.Type(ixRhs)==this.Quantity.Type(ii)
337                 oldValue = model.Variant.getQuantity(this.Variant, ii, ':');
338                 newValue = model.Variant.getQuantity(rhs.Variant, ixRhs, ':');
339                 if flags.Growth
340                     newValue = real(oldValue) + 1i*imag(newValue);
341                 elseif flags.Level
342                     newValue = real(newValue) + 1i*imag(oldValue);
343                 end
344                 this.Variant = model.Variant.assignQuantity( ...
345                     this.Variant, ii, ':', newValue ...
346                     );
347                 ixQty(ii) = true;
348             else
349                 ixMatchingType(ii) = false;
350             end
351         end
352         lsStdCorr = [ getStdName(this.Quantity), getCorrName(this.Quantity) ];
353         lsStdCorrRhs = [ getStdName(rhs.Quantity), getCorrName(rhs.Quantity) ];
354         for ii = 1 : length(lsStdCorr)
355             ixRhs = strcmpi(lsStdCorr{ii}, lsStdCorrRhs);
356             if ~any(ixRhs)
357                 continue
358             end
359             newValue = model.Variant.getStdCorr(rhs.Variant, ixRhs, ':');
360             this.Variant = model.Variant.assignStdCorr( ...
361                 this.Variant, ii, ':', newValue, ...
362                 this.Quantity.IxStdCorrAllowed ...
363                 );
364             ixStdCorr(ii) = true;
365         end
366         if any(~ixMatchingType)
367             utils.warning('modelobj:assign', ...
368                 'This name not assigned because of type mismatch: %s ', ...
369                 this.Quantity.Name{~ixMatchingType});
370         end
371     end

376     function assignNameAndValue(name, posQty, posStdCorr, value)
377         % One or more names, one value.
378         if isempty(posQty) && isempty(posStdCorr)
379             ell = lookup(this.Quantity, name);
380             ixQty = ell.IxName;
381             ixStdCorr = ell.IxStdCorr;
382             if ~any(ixQty) && ~any(ixStdCorr)
383                 if ~isa(lsAllName{i}, 'rexp')
384                     lsInvalidLhsName{end+1} = name;
385                 end
386                 return
387             end
388             posQty = find(ixQty);
389             posStdCorr = find(ixStdCorr);
390         end
391         value = value(:);
392         value = permute(value, [2, 3, 1]);
393         nValue = numel(value);
394         isValidLen = nValue==1 || nValue==nAlt;
395         if ~isValidLen
396             lsInvalidLen{end+1} = name;
397             return
398         end
399         isValidImag = all(imag(value)==0) ...
400             || ( ~flags.Growth && ~flags.Level ...
401             && all(isnan(posStdCorr)) ...
402             && ~any(this.Quantity.Type(posQty)==TYPE(4)) );
403         if ~isValidImag
404             lsInvalidImag{end+1} = name;
405             return
406         end
407         % Assign Variant{:}.Quantity
408         for pos = posQty(~isnan(posQty))
409             if flags.Level || flags.Growth
410                 x = model.Variant.getQuantity( ...
411                     this.Variant, pos, ':' ...
412                     );
413                 if flags.Growth
414                     value = real(x) + 1i*value;
415                 elseif flags.Level
416                     value = value + 1i*imag(x);
417                 end
418             end
419             this.Variant = model.Variant.assignQuantity( ...
420                 this.Variant, pos, ':', value ...
421                 );
422         end
423         % Assign Variant{:}.StdCorr
424         for pos = posStdCorr(~isnan(posStdCorr))
425             this.Variant = model.Variant.assignStdCorr( ...
426                 this.Variant, pos, ':', value, ...
427                 this.Quantity.IxStdCorrAllowed ...
428                 );
429         end
430     end

435     function assignListAndValue( )
436         if size(value,2)==1 && nName>1
437             value = repmat(value, 1, nName, 1);
438         end
439         if size(value,3)==1 && nAlt>1
440             value = repmat(value, 1, 1, nAlt);
441         end
442         if (flags.Growth || flags.Level) && any(imag(value(:))~=0)
443             utils.error('modelobj:assign', ...
444                 ['Cannot assign(...) non-zero imag numbers ', ...
445                 'with flag ''-level'' or ''-growth''.']);
446         end
447         if flags.Level || flags.Growth
448             x = model.Variant.getQuantity(this.Variant, posQty, ':');
449             if flags.Growth
450                 value(1, ixQtyRhs, :) = real(x) + 1i*value(1, ixQtyRhs, :);
451             elseif flags.Level
452                 value(1, ixQtyRhs, :) = value(1, ixQtyRhs, :) + 1i*imag(x);
453             end
454         end
455         if any(ixQtyRhs)
456             ixQty(posQty) = true;
457             this.Variant = model.Variant.assignQuantity( ...
458                 this.Variant, posQty, ':', value(1, ixQtyRhs, :) ...
459                 );
460         end
461         if any(ixStdCorrRhs)
462             ixStdCorr(posStdCorr) = true;
463             this.Variant = model.Variant.assignStdCorr( ...
464                 this.Variant, posStdCorr, ':', value(1, ixStdCorrRhs, :), ...
465                 this.Quantity.IxStdCorrAllowed ...
466                 );
467         end
468     end

473     function chkValid( )
474         if ~isempty(lsInvalidLen)
475             utils.error( 'modelobj:assign', ...
476                 ['Incorrect number of alternative values assigned ', ...
477                 'to this name: %s.'], ...
478                 lsInvalidLen{:} );
479         end
480         if ~isempty(lsInvalidImag)
481             utils.error( 'modelobj:assign', ...
482                 'This name cannot be assigned complex or imag number: %s ', ...
483                 lsInvalidImag{:} );
484         end
485     end

490     function chkLhsNames( )
491         if ~isempty(lsInvalidLhsName)
492             throw( exception.Base('Model:INVALID_NAME', 'error'), ...
493                 '', lsInvalidLhsName{:} ); %#ok<GTARG>
494         end
495     end

500     function listAssigned( )
501         lsAssigned = this.Quantity.Name(ixQty);
502         lse = this.Quantity.Name(ixe);
503         lsAssigned = [lsAssigned, strcat('std_', lse(ixStdCorr(1:ne)))];
504         pos = find( tril(ones(ne),-1)==1 );
505         temp = zeros(ne);
506         temp(pos(ixStdCorr(ne+1:end))) = 1;
507         [row,col] = find(temp==1);
508         lsAssigned = [ lsAssigned, ...
509             strcat('corr_', lse(row), '__', lse(col)) ];
510     end
