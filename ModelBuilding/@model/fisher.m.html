1 function [F, FF, delta, freq, G, step] = fisher(this, nPer, lsPar, varargin)
  2 % fisher  Approximate Fisher information matrix in frequency domain.
  3 %
  4 % __Syntax__
  5 %
  6 %     [F, FF, Delta, Freq] = fisher(M, NPer, PList, ...)
  7 %
  8 %
  9 % __Input Arguments__
 10 %
 11 % * `M` [ model ] - Solved model object.
 12 %
 13 % * `NPer` [ numeric ] - Length of the hypothetical range for which the
 14 % Fisher information will be computed.
 15 %
 16 % * `PList` [ cellstr ] - List of parameters with respect to which the
 17 % likelihood function will be differentiated.
 18 %
 19 %
 20 % __Output Arguments__
 21 %
 22 % * `F` [ numeric ] - Approximation of the Fisher information matrix.
 23 %
 24 % * `FF` [ numeric ] - Contributions of individual frequencies to the total
 25 % Fisher information matrix.
 26 %
 27 % * `Delta` [ numeric ] - Kronecker delta by which the contributions in
 28 % `Fi` need to be multiplied to sum up to `F`.
 29 %
 30 % * `Freq` [ numeric ] - Vector of frequencies at which the Fisher
 31 % information matrix is evaluated.
 32 %
 33 %
 34 % __Options__
 35 %
 36 % * `'ChkSstate='` [ `true` | *`false`* | cell ] - Check steady state in
 37 % each iteration; works only in non-linear models.
 38 %
 39 % * `'Deviation='` [ *`true`* | `false` ] - Exclude the steady state effect
 40 % at zero frequency.
 41 %
 42 % * `'Exclude='` [ char | cellstr | *empty* ] - List of measurement
 43 % variables that will be excluded from the likelihood function.
 44 %
 45 % * `'Percent='` [ `true` | *`false`* ] - Report the overall Fisher matrix
 46 % `F` as Hessian w.r.t. the log of variables; the interpretation for this
 47 % is that the Fisher matrix describes the changes in the log-likelihood
 48 % function in reponse to percent, not absolute, changes in parameters.
 49 %
 50 % * `'Progress='` [ `true` | *`false`* ] - Display progress bar in the
 51 % command window.
 52 %
 53 % * `'Solve='` [ *`true`* | `false` | cellstr ] - Re-compute solution in
 54 % each differentiation step; you can specify a cell array with options for
 55 % the `solve` function.
 56 %
 57 % * `'Sstate='` [ `true` | *`false`* | cell ] - Re-compute steady state in
 58 % each differentiation step; if the model is non-linear, you can pass in a
 59 % cell array with opt used in the `sstate( )` function.
 60 %
 61 %
 62 % __Description__
 63 %
 64 %
 65 % __Example__
 66 %
 67 
 68 % -IRIS Macroeconomic Modeling Toolbox.
 69 % -Copyright (c) 2007-2017 IRIS Solutions Team.
 70 
 71 TYPE = @int8;
 72 
 73 persistent INPUT_PARSER
 74 if isempty(INPUT_PARSER)
 75     INPUT_PARSER = extend.InputParser('model/fisher');
 76     INPUT_PARSER.addRequired('M', @(x) isa(x, 'model'));
 77     INPUT_PARSER.addRequired('NPer', @(x) isnumeric(x) && numel(x)==1 && x==round(x) && x>0);
 78     INPUT_PARSER.addRequired('PList', @(x) iscellstr(x) || ischar(x));
 79 end
 80 INPUT_PARSER.parse(this, nPer, lsPar);
 81 
 82 % Read and validate optional input arguments.
 83 opt = passvalopt('model.fisher', varargin{:});
 84 
 85 ixy = this.Quantity.Type==TYPE(1);
 86 [ny, ~, ~, nf, ne] = sizeOfSolution(this.Vector);
 87 nv = length(this);
 88 
 89 % Process the 'exclude' option.
 90 ixExclude = false(ny, 1);
 91 if ~isempty(opt.exclude)
 92     if ischar(opt.exclude)
 93         opt.exclude = regexp(opt.exclude, '\w+', 'match');
 94     end
 95     lsMeasurementVar = this.Quantity.Name(ixy);
 96     for i = 1 : length(opt.exclude)
 97         ix = strcmp(lsMeasurementVar, opt.exclude{i});
 98         ixExclude(ix) = true;
 99     end
100 end
101 
102 % Get parameter cellstr list from a char list.
103 if ischar(lsPar)
104     lsPar = regexp(lsPar, '\w+', 'match');
105 end
106 
107 % Initialise steady-state solver and chksstate options.
108 opt.Steady = prepareSteady(this, 'silent', opt.sstate);
109 opt.chksstate = prepareChkSteady(this, 'silent', opt.chksstate);
110 opt.solve = prepareSolve(this, 'silent, fast', opt.solve);
111 
112 EPSILON = eps( )^opt.epspower;
113 
114 %--------------------------------------------------------------------------
115 
116 ny = ny - sum(ixExclude);
117 if ny==0
118     utils.warning('model:fisher', ...
119         'No measurement variables included in computing Fisher matrix.');
120 end
121 
122 ixYLog = this.Quantity.IxLog(ixy);
123 ixYLog(ixExclude) = [ ];
124 
125 ell = lookup(this.Quantity, lsPar, TYPE(4));
126 posQty = ell.PosName;
127 posStdCorr = ell.PosStdCorr;
128 
129 ixAssignNan = isnan(posQty);
130 ixStdcorrNan = isnan(posStdCorr);
131 ixValid = ~ixAssignNan | ~ixStdcorrNan;
132 if any(~ixValid)
133     utils.error('model:fisher', ...
134         'This is not a valid parameter name: ''%s''.', ...
135         lsPar{~ixValid});
136 end
137 
138 itr = model.IterateOver( );
139 itr.Quantity = this.Variant.Values;
140 itr.StdCorr = this.Variant.StdCorr;
141 itr.PosQty = posQty;
142 itr.PosStdCorr = posStdCorr;
143 
144 nPList = length(lsPar);
145 nFreq = floor(nPer/2) + 1;
146 freq = 2*pi*(0 : nFreq-1)/nPer;
147 
148 % Kronecker delta vector.
149 % Different for even or odd number of periods.
150 delta = ones(1, nFreq);
151 if mod(nPer, 2)==0
152     delta(2:end-1) = 2;
153 else
154     delta(2:end) = 2;
155 end
156 
157 FF = nan(nPList, nPList, nFreq, nv);
158 F = nan(nPList, nPList, nv);
159 
160 % Create a command-window progress bar.
161 if opt.progress
162     progress = ProgressBar('IRIS model.fisher progress');
163 end
164 
165 throwErr = true;
166 
167 for v = 1 : nv
168     % Fetch the i-th parameterisation.
169     m = this(v);
170 
171     % Minimum necessary state space.
172     [T0, R0, Z0, H0, Omg0, nUnit0] = getSspace( );
173 
174     % SGF and inverse SGF at p0.
175     [G, Gi] = computeSgfy(T0, R0, Z0, H0, Omg0, nUnit0, freq, opt);
176 
177     % Compute derivatives of SGF and steady state
178     % wrt the selected parameters.
179     dG = nan(ny, ny, nFreq, nPList);
180     if ~opt.deviation
181         dy = zeros(ny, nPList);
182     end
183     % Determine differentiation step.
184     p0 = nan(1, nPList);
185     p0(~ixAssignNan) = m.Variant.Values(:, posQty(~ixAssignNan), :);
186     p0(~ixStdcorrNan) = m.Variant.StdCorr(:, posStdCorr(~ixStdcorrNan), :);
187     step = max([abs(p0);ones(1, nPList)], [ ], 1)*EPSILON;
188 
189     for i = 1 : nPList
190         pp = p0;
191         pm = p0;
192         pp(i) = pp(i) + step(i);
193         pm(i) = pm(i) - step(i);
194         twoSteps = pp(i) - pm(i);
195 
196         isSstate = ~opt.deviation && ~isnan(posQty(i));
197 
198         % Steady state, state space and SGF at p0(i) + step(i).
199         m = update(m, pp, itr, 1, opt, throwErr);
200         if isSstate
201             yp = getSstate( );
202         end
203         [Tp, Rp, Zp, Hp, Omgp, nUnitp] = getSspace( );
204         Gp = computeSgfy(Tp, Rp, Zp, Hp, Omgp, nUnitp, freq, opt);
205 
206         % Steady state, state space and SGF at p0(i) - step(i).
207         m = update(m, pm, itr, 1, opt, throwErr);
208         if isSstate
209             ym = getSstate( );
210         end
211         [Tm, Rm, Zm, Hm, Omgm, nUnitm] = getSspace( );
212         Gm = computeSgfy(Tm, Rm, Zm, Hm, Omgm, nUnitm, freq, opt);
213 
214         % Differentiate SGF and steady state.
215         dG(:, :, :, i) = (Gp - Gm) / twoSteps;
216         if isSstate
217             dy(:, i) = real(yp(:) - ym(:)) / twoSteps;
218         end
219 
220         % Reset model parameters to `p0`.
221         m.Variant.Values(:, posQty(~ixAssignNan), :) = p0(1, ~ixAssignNan);
222         m.Variant.StdCorr(:, posStdCorr(~ixStdcorrNan), :) = p0(1, ~ixStdcorrNan);
223 
224         % Update the progress bar.
225         if opt.progress
226             update(progress, ((v-1)*nPList+i)/(nv*nPList));
227         end
228 
229     end
230 
231     % Compute Fisher information matrix.
232     % Steady-state-independent part.
233     for i = 1 : nPList
234         for j = i : nPList
235             fi = zeros(1, nFreq);
236             for k = 1 : nFreq
237                 fi(k) = ...
238                     trace(real(Gi(:, :, k)*dG(:, :, k, i)*Gi(:, :, k)*dG(:, :, k, j)));
239             end
240             if ~opt.deviation
241                 % Add steady-state effect to zero frequency.
242                 % We don't divide the effect by 2*pi because
243                 % we skip dividing G by 2*pi, too.
244                 A = dy(:, i)*dy(:, j)';
245                 fi(1) = fi(1) + nPer*trace(Gi(:, :, 1)*(A + A'));
246             end
247             FF(i, j, :, v) = fi;
248             FF(j, i, :, v) = fi;
249             f = delta*fi';
250             F(i, j, v) = f;
251             F(j, i, v) = f;
252         end
253     end
254 
255     if opt.percent
256         P0 = diag(p0);
257         F(:, :, v) = P0*F(:, :, v)*P0;
258     end
259 
260 end
261 % End of main loop.
262 
263 FF = FF / 2;
264 F = F / 2;
265 
266 return
267 
268 
269     function [T, R, Z, H, Omg, numOfUnitRoots] = getSspace( )
270         T = m.Variant.Solution{1};
271         numOfUnitRoots = getNumOfUnitRoots(m.Variant);
272         Z = m.Variant.Solution{4}(~ixExclude, :);
273         T = T(nf+1:end, :);
274         % Cut off forward expansion.
275         R = m.Variant.Solution{2}(nf+1:end, 1:ne);
276         H = m.Variant.Solution{5}(~ixExclude, 1:ne);
277         Omg = omega(m);
278     end
279 
280 
281     function y = getSstate( )
282         % Get the steady-state levels for the measurement variables.
283         y = m.Variant.Values(:, ixy, :);
284         y = real(y);
285         % Adjust for the excluded measurement variables.
286         y(ixExclude) = [ ];
287         % Take log of log variables; `ixYLog` has been already adjusted
288         % for the excluded measurement variables.
289         y(ixYLog) = log(y(ixYLog));
290     end
291 end

294 function [G, Gi] = computeSgfy(T, R, Z, H, Omg, numOfUnitRoots, freq, opt)
295     % Spectrum generating function and its inverse.
296     % Computationally optimised for observables.
297     [ny, nb] = size(Z);
298     nFreq = length(freq(:));
299     Sgm1 = R*Omg*R.';
300     Sgm2 = H*Omg*H.';
301     G = nan(ny, ny, nFreq);
302     for i = 1 : nFreq
303         iFreq = freq(i);
304         if iFreq==0 && numOfUnitRoots>0
305             % Exclude the unit-root part of the transition matrix, and compute SGF only
306             % for the stable part. Stationary variables are unaffected.
307             Z0 = Z(:, numOfUnitRoots+1:end);
308             T0 = T(numOfUnitRoots+1:end, numOfUnitRoots+1:end);
309             R0 = R(numOfUnitRoots+1:end, :);
310             X = Z0 / (eye(nb-numOfUnitRoots) - T0);
311             G(:, :, i) = trimSymmetric(X*(R0*Omg*R0.')*X' + Sgm2);
312         else
313             X = Z/(eye(nb) - T*exp(-1i*iFreq));
314             G(:, :, i) = trimSymmetric(X*Sgm1*X' + Sgm2);
315         end
316 
317     end
318     % Do not divide G by 2*pi.
319     % First, this cancels out in Gi*dG*Gi*dG
320     % and second, we do not divide the steady-state effect
321     % by 2*pi either.
322     if nargout>1
323         Gi = nan(ny, ny, nFreq);
324         if opt.chksgf
325             for i = 1 : nFreq
326                 Gi(:, :, i) = computePseudoInverse(G(:, :, i), opt.tolerance);
327             end
328         else
329             for i = 1 : nFreq
330                 Gi(:, :, i) = inv(G(:, :, i));
331             end
332         end
333     end
334 end

337 function x = trimSymmetric(x)
338     % Minimise numerical inaccuracy between upper and lower parts
339     % of symmetric matrices.
340     ix = eye(size(x))==1;
341     x = (x + x.')/2;
342     x(ix) = real(x(ix));
343 end

346 function X = computePseudoInverse(A, Tol)
347     c = class(A);
348     if isempty(A)
349         X = zeros(size(A'), c);
350         return
351     end
352     sizeOfA = size(A);
353     s = svd(A);
354     r = sum(s/s(1)>Tol);
355     if r==0
356         X = zeros(size(A'), c);
357     elseif r==sizeOfA(1)
358         X = inv(A);
359     else
360         [U, ~, V] = svd(A, 0);
361         S = diag(1./s(1:r));
362         X = V(:, 1:r)*S*U(:, 1:r)';
363     end
364 end
